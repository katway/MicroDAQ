CREATE VIEW [dbo].[modubs_control] AS SELECT   m.Code, v.command,  v.cycle,  de.Slave, de.SerialID, rt.RegisterName, c.RegesiterAddress, c.Length, v.typeFROM         dbo.v_remoteControl AS v LEFT OUTER JOIN                      dbo.MetaData AS m ON v.id = m.Code LEFT OUTER JOIN                      dbo.ModbusCommands AS c ON m.ModbusCommands_SerialID = c.SerialID LEFT OUTER JOIN                      dbo.RegisterType AS rt ON c.RegisterType = rt.serialId LEFT OUTER JOIN                      dbo.ModbusSlave AS de ON c.ModbusSlave_SerialID = de.SerialIDGO
